9/// Account manager for handling chart of accounts operations
10pub struct AccountManager<S: LedgerStorage> {
11    pub(crate) storage: S,
12    validator: Box<dyn AccountValidator>,
13}
14
15impl<S: LedgerStorage> AccountManager<S> {
16    /// Create a new account manager
17    pub fn new(storage: S) -> Self {
18        Self {
19            storage,
20            validator: Box::new(DefaultAccountValidator),
21        }
22    }
23
24    /// Create a new account manager with custom validator
25    pub fn with_validator(storage: S, validator: Box<dyn AccountValidator>) -> Self {
26        Self { storage, validator }
27    }
28
29    /// Create a new account
30    pub async fn create_account(
31        &mut self,
32        id: String,
33        name: String,
34        account_type: AccountType,
35        parent_id: Option<String>,
36    ) -> LedgerResult<Account> {
37        let account = Account::new(id, name, account_type, parent_id);
38
39        // Validate the account
40        self.validator.validate_account(&account)?;
41
42        // Check if account already exists
43        if let Some(_existing) = self.storage.get_account(&account.id).await? {
44            return Err(LedgerError::Validation(format!(
45                "Account with ID '{}' already exists",
46                account.id
47            )));
48        }
49
50        // Validate parent account exists if specified
51        if let Some(ref parent_id) = account.parent_id {
52            if self.storage.get_account(parent_id).await?.is_none() {
53                return Err(LedgerError::Validation(format!(
54                    "Parent account '{}' does not exist",
55                    parent_id
56                )));
57            }
58        }
59
60        // Save the account
61        self.storage.save_account(&account).await?;
62
63        Ok(account)
64    }
65
66    /// Get an account by ID
67    pub async fn get_account(&self, account_id: &str) -> LedgerResult<Option<Account>> {
68        self.storage.get_account(account_id).await
69    }
70
71    /// Get an account by ID, returning an error if not found
72    pub async fn get_account_required(&self, account_id: &str) -> LedgerResult<Account> {
73        self.storage
74            .get_account(account_id)
75            .await?
76            .ok_or_else(|| LedgerError::AccountNotFound(account_id.to_string()))
77    }
78
79    /// List all accounts
80    pub async fn list_accounts(&self) -> LedgerResult<Vec<Account>> {
81        self.storage.list_accounts(None).await
82    }
83
84    /// List accounts by type
85    pub async fn list_accounts_by_type(
86        &self,
87        account_type: AccountType,
88    ) -> LedgerResult<Vec<Account>> {
89        self.storage.list_accounts(Some(account_type)).await
90    }
91
92    /// Update an account
93    pub async fn update_account(&mut self, account: &Account) -> LedgerResult<()> {
94        // Validate the account
95        self.validator.validate_account(account)?;
96
97        // Ensure the account exists
98        if self.storage.get_account(&account.id).await?.is_none() {
99            return Err(LedgerError::AccountNotFound(account.id.clone()));
100        }
101
102        self.storage.update_account(account).await
103    }
104
105    /// Delete an account
106    pub async fn delete_account(&mut self, account_id: &str) -> LedgerResult<()> {
107        // Validate deletion
108        self.validator.validate_account_deletion(account_id)?;
109
110        // Ensure the account exists
111        if self.storage.get_account(account_id).await?.is_none() {
112            return Err(LedgerError::AccountNotFound(account_id.to_string()));
113        }
114
115        self.storage.delete_account(account_id).await
116    }
117
118    /// Get account balance
119    pub async fn get_balance(
120        &self,
121        account_id: &str,
122        as_of_date: Option<chrono::NaiveDate>,
123    ) -> LedgerResult<BigDecimal> {
124        self.storage
125            .get_account_balance(account_id, as_of_date)
126            .await
127    }
128}

130/// Chart of accounts implementation
131pub struct StandardChartOfAccounts<S: LedgerStorage> {
132    account_manager: AccountManager<S>,
133}
134
135impl<S: LedgerStorage> StandardChartOfAccounts<S> {
136    /// Create a new chart of accounts
137    pub fn new(storage: S) -> Self {
138        Self {
139            account_manager: AccountManager::new(storage),
140        }
141    }
142}
143
144#[async_trait::async_trait]
145impl<S: LedgerStorage> ChartOfAccounts for StandardChartOfAccounts<S> {
146    async fn get_chart(&self) -> LedgerResult<Vec<Account>> {
147        self.account_manager.list_accounts().await
148    }
149
150    async fn add_account(&mut self, account: Account) -> LedgerResult<()> {
151        self.account_manager.storage.save_account(&account).await
152    }
153
154    async fn get_child_accounts(&self, parent_id: &str) -> LedgerResult<Vec<Account>> {
155        let all_accounts = self.account_manager.list_accounts().await?;
156        Ok(all_accounts
157            .into_iter()
158            .filter(|account| account.parent_id.as_deref() == Some(parent_id))
159            .collect())
160    }
161
162    async fn get_account_path(&self, account_id: &str) -> LedgerResult<Vec<Account>> {
163        let mut path = Vec::new();
164        let mut current_account_id = Some(account_id.to_string());
165
166        while let Some(id) = current_account_id {
167            match self.account_manager.get_account(&id).await? {
168                Some(account) => {
169                    current_account_id = account.parent_id.clone();
170                    path.insert(0, account);
171                }
172                None => {
173                    return Err(LedgerError::AccountNotFound(id));
174                }
175            }
176        }
177
178        Ok(path)
179    }
180}
